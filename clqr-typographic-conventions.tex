\section*{Typographic Conventions}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{LIST}{1cm}

  \IT{\kwd{NAME}; \FU{NAME}; \GFU{NAME}; \MC{NAME}; \SO{NAME}; \V{\A name\A};
    \CNS{name}}
  {
  Symbol defined in Common Lisp; esp.\ function, generic function,
  macro, special operator, variable, constant.
  }

  \IT{\VAR{them}\qquad\qquad\qquad}
  {
  Placeholder for actual code.
  }

  \IT{\LIT{me}\qquad\qquad\qquad\qquad}
  {
  Literal text.
  }

  \IT{\Op{\VAR{foo}\DF{\LIT{bar}}}\qquad\qquad}
  {
  Either one \VAR{foo} or nothing; defaults to \LIT{bar}.
  }

  \IT{\OPn{\VAR{foo}}; \Goos{\VAR{foo}}}
  {
  Zero or more \VAR{foo}s.
  }

  \IT{\RP{\VAR{foo}}; \Goop{\VAR{foo}}}
  {
  One or more \VAR{foo}s.
  }

  \IT{\VAR{foos}\qquad\qquad\qquad}
  {
  English plural denotes a list argument.
  }

  \IT{\Goo{\VAR{foo}\XOR\VAR{bar}\XOR\VAR{baz}};
    \xorGOO{
      \VAR{foo}\\
      \VAR{bar}\\
      \VAR{baz}%
    }{.}}
  {
  Either \VAR{foo}, or \VAR{bar}, or \VAR{baz}.
  }

  \IT{\orGOO{
      \VAR{foo}\\
      \VAR{bar}\\
      \VAR{baz} 
    }{.}}
  {
  Anything from none to each of \VAR{foo}, \VAR{bar}, and \VAR{baz}.
  }

  \IT{\NEV{\VAR{foo}}\qquad\qquad\qquad}
  {
  Argument \VAR{foo} is not evaluated.
  }

  \IT{\DES{\VAR{bar}}\qquad\qquad\qquad}
  {
  Argument \VAR{bar} is possibly modified.
  }

  \IT{\PROGN{\VAR{foo}}\qquad\qquad\qquad}
  {
  \OPn{\VAR{foo}} is evaluated as in \SO{progn}; see page \pageref{:progn}.
  }

  \IT{\retval{\VAR{foo}}; \retvalii{\VAR{bar}};
    \retvaln{\VAR{n}}{\VAR{baz}}}
  {
  Primary, secondary, and \VAR{n}th return value.
  }

  \IT{\T; \NIL\qquad\qquad\qquad}
  {
  \kwd*{T}, or truth in general; and \kwd*{NIL} or \kwd{()}.
  }
  
\end{LIST}
